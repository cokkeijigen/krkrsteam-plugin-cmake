cmake_minimum_required(VERSION 3.20.0)
project(krkrsteam)

set(CMAKE_CXX_STANDARD 20)

set(SOURCE_DIR "${CMAKE_CURRENT_LIST_DIR}/src")
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/$<CONFIG>")
set(PROJECT_LIBRARS_DIRECTORY "${CMAKE_CURRENT_LIST_DIR}/libs")

add_compile_options("/source-charset:utf-8")

add_subdirectory("${PROJECT_LIBRARS_DIRECTORY}/ncbind")
add_subdirectory("${PROJECT_LIBRARS_DIRECTORY}/steamapi")

file(GLOB_RECURSE SOURCES "${SOURCE_DIR}/*.cpp")
add_library(${PROJECT_NAME} SHARED ${SOURCES})

target_include_directories(${PROJECT_NAME} PUBLIC ${SOURCE_DIR})

target_link_libraries(${PROJECT_NAME} ncbind steamapi)

target_copy_steam_api_to_ouput_dir(${PROJECT_NAME})
